custom animate



	const element = new Razonimate('.div1', {
	  backgroundColor: 'red',
	  opacity: 1,
	  textColor: 'white',
	  duration: 2000,
	  easing: 'ease-in-out',
	  offsetX: -100,
	  offsetY: 0,
	  scale: 1
	});


initialize razonimate



	const razonimate = new Razonimate();
	razonimate.init();
